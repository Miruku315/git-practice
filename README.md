```cpp

using namespace std;
int main(){
  int x;
  cin >> x;
  switch (x/10){
    case 10:
      cout << "exellent";
      break;
    case 9:
      cout << "very good";
      break;
    default:
      cout << "ok";
    }
  return 0;
}
```
